# x86/x64

## 概述

x86 是 Intel 和 AMD 處理器使用的指令集架構，從 1978 年的 Intel 8086 發展而來。x86-64（又稱 amd64）是 64 位元擴展，是目前個人電腦和伺服器最廣泛使用的處理器架構。

## 歷史

- **1978**：Intel 8086 發布，16 位元
- **1985**：Intel 80386，32 位元
- **1999**：AMD 引入 x86-64
- **2004**：Intel 引入 EM64T
- **現在**：x86-64 為主流

## 暫存器

### 1. x86-64 暫存器

```asm
; 通用暫存器（64-bit）
rax, rbx, rcx, rdx
rsi, rdi, rbp, rsp
r8, r9, r10, r11
r12, r13, r14, r15

; 32-bit 部分
eax, ebx, ecx, edx
esi, edi, ebp, esp
r8d, r9d, ...

; 16-bit 部分
ax, bx, cx, dx
si, di, bp, sp
r8w, ...

; 8-bit 部分
al, ah, bl, bh, cl, ch, dl, dh
r8b, r9b, ...

; 特殊暫存器
rip  ; 指令指標
rflags ; 標誌暫存器
```

### 2. 基本指令

```asm
; 資料傳輸
mov rax, rbx      ; rax = rbx
mov eax, 42       ; eax = 42
mov [rsi], rax    ; 記憶體 = rax
mov rax, [rdi]    ; rax = 記憶體

; 算術運算
add rax, rbx      ; rax += rbx
sub rax, rbx      ; rax -= rbx
imul rax, rbx     ; rax *= rbx
idiv rbx          ; rax /= rbx (edx:eax)

inc rax           ; rax++
dec rax           ; rax--

; 邏輯運算
and rax, rbx      ; rax &= rbx
or rax, rbx       ; rax |= rbx
xor rax, rax      ; rax = 0（常用技巧）
not rax           ; rax = ~rax

; 比較
cmp rax, rbx      ; 設定標誌
test rax, rax     ; 測試是否為 0
```

### 3. 控制流

```asm
; 跳轉
jmp label         ; 無條件跳轉
je label          ; 等於跳轉
jne label         ; 不等於跳轉
jz label          ; 為 0 跳轉
jnz label         ; 不為 0 跳轉

; 條件跳轉
jl label          ; 小於跳轉
jle label         ; 小於等於跳轉
jg label          ; 大於跳轉
jge label         ; 大於等於跳轉

; 函數呼叫
call func         ; 呼叫函數
ret               ; 返回

; 範例：if (a > b)
cmp rax, rbx
jle skip
; do something
skip:
```

### 4. 函數框架

```asm
; 函數 prologue
push rbp
mov rbp, rsp
sub rsp, 16       ; 配置局部變數

; 函數 epilogue
mov rsp, rbp
pop rbp
ret

; 系統呼叫
mov rax, 60       ; exit 系統呼叫
mov rdi, 0        ; 退出碼
syscall
```

### 5. 完整範例

```asm
section .data
    fmt db "Hello, %d", 10, 0

section .text
    global main

main:
    push rbp
    mov rbp, rsp
    
    mov rax, 5
    mov rbx, 3
    add rax, rbx
    
    ; printf("%d\n", rax)
    lea rdi, [rel fmt]
    mov rsi, rax
    xor eax, eax
    call printf
    
    xor eax, eax
    pop rbp
    ret
```

### 6. SIMD 指令

```asm
; SSE/SSE2
movaps xmm0, [rsi]    ; 載入 128-bit 對齊資料
addps xmm0, xmm1      ; 浮點加法
mulps xmm0, xmm2      ; 浮點乘法

; AVX/AVX2
vaddps ymm0, ymm1, ymm2  ; 256-bit 浮點加法
vmovups zmm0, [rsi]      ; 512-bit 載入
```

## 系統呼叫

```asm
; Linux x86-64 系統呼叫
; 系統呼叫號在 rax
; 參數: rdi, rsi, rdx, r10, r8, r9
; 返回值在 rax

; write(fd, buf, count)
mov rax, 1        ; write
mov rdi, 1        ; stdout
lea rsi, [rel msg]
mov rdx, 13       ; length
syscall

; exit(code)
mov rax, 60       ; exit
mov rdi, 0        ; status
syscall

section .data
msg db "Hello World", 10
```

## 為什麼學習 x86？

3. **系統程式設計**：驅動程式開發
4. **惡意程式分析**：安全研究
